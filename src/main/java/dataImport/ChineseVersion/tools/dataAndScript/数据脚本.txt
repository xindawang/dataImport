CREATE TABLE DATADIC.DATA_DIC_CHN_XML
(
    ID INT PRIMARY KEY,
    DATA_ELEMENT_NAME VARCHAR2(100),
    SIMPLE_OR_COMPOUND VARCHAR2(30),
    DATA_TYPE VARCHAR2(20),
    MIN_LEN NUMBER,
    MAX_LEN NUMBER,
    MIN_VAL NUMBER,
    MAX_VAL NUMBER
);


CREATE TABLE DATADIC.DATA_DIC_CHN_SUB
(
    ID INT PRIMARY KEY,
    SUPER_DATA_ELEMENT VARCHAR2(100),
    DATA_ELEMENT_NAME VARCHAR2(100),
    REQUIED VARCHAR2(15),
    REPEATABLE VARCHAR2(10)
);

CREATE TABLE DATADIC.DATA_DIC_CHN
(
    ID INT PRIMARY KEY,
    DATA_ELEMENT_NAME VARCHAR2(100),
    TEI_OR_ACRONYM VARCHAR2(4),
    FORMAT VARCHAR2(10),
	XML_DATA_TYPE VARCHAR2(16),
    SUB_DATA_ELEMENTS VARCHAR2(20),
    ATTRIBUTES VARCHAR2(30),
    USAGE VARCHAR2(150),
    DESCRIPTION_OR_PURPOSE VARCHAR2(600),
    CODES VARCHAR2(1500),
    REMARKS CLOB,
    EXAMPLES CLOB
);

CREATE TABLE DATADIC.DATA_DIC_CHN_CODE
(
    ID INT PRIMARY KEY,
    DATA_ELEMENT_NAME VARCHAR2(100),
    CODE_NAME VARCHAR2(20),
    DESCRIPTION VARCHAR2(300),
    SPECIFICATION VARCHAR2(500)
);
