DROP  SEQUENCE DATA_DIC_ENG_ID_SEQ;
CREATE SEQUENCE DATA_DIC_ENG_ID_SEQ MINVALUE 1 NOMAXVALUE INCREMENT BY 1 START WITH 1 NOCACHE;
CREATE OR REPLACE TRIGGER DATA_DIC_ENG_ID_SEQ_INS_TRG BEFORE INSERT ON DATA_DIC_ENG FOR EACH ROW WHEN(NEW.ID IS NULL)
  BEGIN
    SELECT DATA_DIC_ENG_ID_SEQ.NEXTVAL INTO :NEW.ID FROM DUAL;
