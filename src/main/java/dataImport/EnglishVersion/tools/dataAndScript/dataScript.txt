CREATE TABLESPACE DATA_DIC
LOGGING
DATAFILE 'D:DATADIC.DBF'
SIZE 4M
AUTOEXTEND ON
NEXT 4M MAXSIZE UNLIMITED
EXTENT MANAGEMENT LOCAL;

ALTER DATABASE
DATAFILE 'D:\DATA.dbf' AUTOEXTEND
ON NEXT 50M MAXSIZE UNLIMITED

CREATE TABLE DATADIC.DATA_DIC_ENG_XML
(
    ID INT PRIMARY KEY,
    DATA_ELEMENT_NAME VARCHAR2(60),
    SIMPLE_OR_COMPOUND VARCHAR2(30),
    DATA_TYPE VARCHAR2(15),
    MIN_LEN NUMBER,
    MAX_LEN NUMBER,
    MIN_VAL NUMBER,
    MAX_VAL NUMBER
);


CREATE TABLE DATADIC.DATA_DIC_ENG_SUB
(
    ID INT PRIMARY KEY,
    SUPER_DATA_ELEMENT VARCHAR2(60),
    DATA_ELEMENT_NAME VARCHAR2(60),
    REQUIED VARCHAR2(10),
    REPEATABLE VARCHAR2(8)
);

CREATE TABLE DATADIC.DATA_DIC_ENG
(
    ID INT PRIMARY KEY,
    DATA_ELEMENT_NAME VARCHAR2(60),
    TEI_OR_ACRONYM VARCHAR2(4),
    FORMAT VARCHAR2(10),
	XML_DATA_TYPE VARCHAR2(16),
    SUB_DATA_ELEMENTS VARCHAR2(20),
    ATTRIBUTES VARCHAR2(20),
    USAGE VARCHAR2(110),
    DESCRIPTION_OR_PURPOSE VARCHAR2(600),
    CODES VARCHAR2(1100),
    REMARKS CLOB,
    EXAMPLES CLOB
);

CREATE TABLE DATADIC.DATA_DIC_ENG_CODE
(
    ID INT PRIMARY KEY,
    DATA_ELEMENT_NAME VARCHAR2(60),
    CODE_NAME VARCHAR2(50),
    DESCRIPTION VARCHAR2(1000),
    SPECIFICATION VARCHAR2(300)
);
